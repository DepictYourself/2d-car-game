176  public void ChangeLane(Car car, Direction direction)

339  private void SpawnBullet(Car car)
340  {

342  {

345  {
346  bullet = new Bullet(
347  car.Position,
348  GameObjectConfig.BulletWidth,
349  GameObjectConfig.BulletHeight,
350  (int)((car.Area.X + (car.Area.Width / 2)) - GameObjectConfig.BulletWidth), // Bullet X center calculation
351  (int)car.Area.Y - 40);
352  }
353  else
354  {
355  bullet = new Bullet(
356  car.Position,
357  GameObjectConfig.BulletWidth,
358  GameObjectConfig.BulletHeight,
359  (int)((car.Area.X + (car.Area.Width / 2)) - GameObjectConfig.BulletWidth), // Bullet X center calculation
360  (int)car.Area.Y + 200);
361  }
362  bullet.Active = true;
363  bullet.Source = car;
364  lock (bulletLockObject)
365  {
366  model.Bullets.Add(bullet);
367  }
368  while (!gameOver && bullet.Active)
369  {
370  if (car.GetType() == typeof(PlayerCar))
371  {
372  bullet.Speed = GameObjectConfig.PlayerBulletSpeed;
373  }
374  else
375  {
376  bullet.Speed = GameObjectConfig.EnemyBulletSpeed;
377  }
378  MoveBullet(bullet);
379  Thread.Sleep(10);
380  }
381  lock (bulletLockObject)
382  {
383  model.Bullets.Remove(bullet);
384  }
385  }, TaskCreationOptions.LongRunning).Start();
386  }
